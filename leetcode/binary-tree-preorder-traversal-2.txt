class Solution:
    def preorderTraversal(self, root: Optional[TreeNode]) -> List[int]:
        if not root: return []

        ans = []
        while stack:
            node = stack.pop()
            ans.append(node.val)
        stack = [root]
            if node.right:
                stack.append(node.right)
            if node.left:
                stack.append(node.left)
        return ans
