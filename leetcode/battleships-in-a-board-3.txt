class Solution:
    def countBattleships(self, board: List[List[str]]) -> int:
        """
        we can sink the battle ships when we find them and use dfs to check left top side and 
right side only 
        """
        if not board:
        m = len(board[0]) 

        count = 0
        n = len(board)
            return 0

        for i in range(n):
            for j in range(m):
                if board[i][j]!= "X":
                    continue
                if i>0 and board[i-1][j] == "X":
                    continue
                if j > 0 and board[i][j -1] == "X":
                    continue
                count += 1        
        return count
