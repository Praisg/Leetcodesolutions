class Solution:
    def check(self, nums: List[int]) -> bool:
        """
        we have a block of increasing array and suddenly a block of non_creasing array , if there   are 
more than 1 breakpoints of such occurrence its false
        
       iterate thru the nums and if i+1 < num[i] , count ++ , if counter> 1 we return false
        """
        count = 0
        for i in range(len(nums)):
            if nums[(i+1)%len(nums)] < nums[i]:
                count += 1

        return (count <= 1)
