class Solution:
    def countBattleships(self, board: List[List[str]]) -> int:
        """
        we can sink the battle ships when we find them and use dfs to check left top side and 
right side only 
        """
        n = len(board)
        m = len(board[0])

        def dfs (i, j):
            if i < 0 or i >= n or j < 0 or j >= m or board[i][j]!="X":
                return
            board[i][j] = "."
            dfs(i+1, j)
            dfs(i, j+1)   


        count = 0
        for i in range(n):
            for j in range(m):
                if board[i][j]== "X":
                    count += 1
                dfs(i, j)
        return count
